      # ========================================= #
      # Alterando o input recebido por uma clula #
      def r_input(lines):
          new_lines = []
          for line in lines:
              chars=line
          # ~~~~~~~~~~~~~~~~~~~~~~~~~#
              if(chars=="%r\n"): # Abertura do cdigo em R / Sinalizao
          # ~~~~~~~~~~~~~~~~~~~~~~~~~#
                  chars='''text="""\n'''
              else:
              # ~~~~~~~~~~~~~~~~~~~~~~~~~#
                  if(chars=="r%\n"): # Fechamento do cdigo em R / Sinalizao
              # ~~~~~~~~~~~~~~~~~~~~~~~~~#
                      chars='''"""\n'''
                      chars=chars+"reval(text)\n"
              new_lines.append(chars)
          return new_lines
